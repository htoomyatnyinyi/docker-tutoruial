version: "3.8"
services:
  e_clients:
    build:
      context: ./e_clients
      dockerfile: Dockerfile
    ports:
      - "5173:5173"
    environment:
      - VITE_API_URL=http://e_server:8800 # Use service name instead of localhost
    depends_on:
      - e_server
    networks:
      - app-network

  e_server:
    build:
      context: ./e_server
      dockerfile: Dockerfile
    ports:
      - "8800:8800"
    environment:
      - NODE_ENV=development
      - DATABASE_URL=mysql://root:hmnn@mysql:3306/mydata # Point to MySQL service
    depends_on:
      - mysql
    networks:
      - app-network

  mysql:
    image: mysql:8.0
    environment:
      - MYSQL_ROOT_PASSWORD=hmnn
      - MYSQL_DATABASE=mydata
    ports:
      - "3306:3306" # Expose MySQL port for local access (optional)
    volumes:
      - mysql-data:/var/lib/mysql
    networks:
      - app-network

networks:
  app-network:
    driver: bridge

volumes:
  mysql-data:
